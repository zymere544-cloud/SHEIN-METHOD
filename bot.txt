<script>
function runBot() {
  const input = document.getElementById("input").value.toLowerCase();
  const output = document.getElementById("output");

  // Example bot commands
  if(input === "hello") output.innerText = "Hello! Welcome to the web bot.";
  else if(input === "info") output.innerText = "This is a web-based bot. You can interact with it here!";
  else if(input === "help") output.innerText = "Commands available: hello, info, help, visit";
  else if(input === "visit") output.innerHTML = 'Click here to visit <a href="https://linkfly.to/YOUR_ACCOUNT" target="_blank">your Linkfly page</a>';
  else output.innerText = "Unknown command. Try 'hello', 'info', 'help', or 'visit'.";
}
